<script>
const form = document.getElementById('playerForm');
const playersBody = document.getElementById('playersBody');
const emptyState = document.getElementById('emptyState');
const toast = document.getElementById('toast');

let players = JSON.parse(localStorage.getItem('footballPlayers')) || [];

function savePlayers() {
    localStorage.setItem('footballPlayers', JSON.stringify(players));
}

function showToast(message) {
    toast.textContent = message;
    toast.classList.add('show');
    setTimeout(() => {
        toast.classList.remove('show');
    }, 3000);
}

function updateStats() {
    const mondayCount = players.filter(p => p.day === 'الاثنين').length;

    document.getElementById('totalPlayers').textContent = players.length;
    document.getElementById('mondayPlayers').textContent = mondayCount;
    document.getElementById('availableSpots').textContent = Math.max(0, 18 - players.length);
}

function renderPlayers() {
    if (players.length === 0) {
        playersBody.innerHTML = '';
        emptyState.style.display = 'block';
        updateStats();
        return;
    }

    emptyState.style.display = 'none';
    playersBody.innerHTML = players.map((player, index) => `
        <tr>
            <td>${index + 1}</td>
            <td>${player.name}</td>
            <td><span class="badge badge-${player.position}">${player.position}</span></td>
            <td>${player.level}</td>
            <td>${player.day || '-'}</td>
            <td>
                <button class="delete-btn" onclick="deletePlayer(${index})">
                    <i class="fas fa-trash"></i>
                </button>
            </td>
        </tr>
    `).join('');

    updateStats();
}

form.addEventListener('submit', (e) => {
    e.preventDefault();

    const newPlayer = {
        name: document.getElementById('name').value,
        phone: document.getElementById('phone').value,
        position: document.getElementById('position').value,
        level: document.getElementById('level').value,
        day: document.getElementById('day').value,
        timestamp: Date.now()
    };

    players.unshift(newPlayer);
    savePlayers();
    renderPlayers();
    form.reset();
    showToast('تم التسجيل بنجاح! ?');
});

function deletePlayer(index) {
    if (confirm('متأكد عايز تحذف اللاعب ده؟')) {
        players.splice(index, 1);
        savePlayers();
        renderPlayers();
        showToast('تم الحذف');
    }
}

renderPlayers();
</script>
